CREATE TABLE POSTS (
	ID VARCHAR  AUTO_INCREMENT PRIMARY KEY,
	PICTURE_BOOKS_ID VARCHAR NOT NULL,      
	TITLE VARCHAR(30),
	GENRE VARCHAR(4) NOT NULL,
	FREE VARCHAR(200),
	PICTURE VARCHAR NOT NULL,
	TIME TIMESTAMP NOT NULL
);
