【テーブルを作成するSQL文】
CREATE TABLE POSTS (
	ID VARCHAR  AUTO_INCREMENT PRIMARY KEY,
	PICTURE_BOOKS_ID VARCHAR NOT NULL,      →(図鑑IDを引っ張ってくる)
	TITLE VARCHAR(30),
	GENRE INTEGER NOT NULL,
	FREE VARCHAR(200),
	PICTURE VARCHAR NOT NULL,
	TIME TIMESTAMP NOT NULL
);
